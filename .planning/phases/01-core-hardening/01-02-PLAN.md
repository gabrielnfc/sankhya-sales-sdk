```typescript
describe('CORE-05: jitter no backoff', () => {
  it('deve aplicar jitter ao delay (full jitter)', async () => {
    const randomSpy = vi.spyOn(Math, 'random').mockReturnValue(0.5);
    const sleepCalls: number[] = [];
    const originalSetTimeout = globalThis.setTimeout;
    vi.spyOn(globalThis, 'setTimeout').mockImplementation((fn: () => void, ms?: number) => {
      if (ms !== undefined && ms > 0) sleepCalls.push(ms);
      return originalSetTimeout(fn, 0);
    });

    const error = new ApiError('server error', '/test', 'GET', 500);
    const fn = vi.fn()
      .mockRejectedValueOnce(error)
      .mockRejectedValueOnce(error)
      .mockResolvedValue('ok');

    await withRetry(fn, { maxRetries: 3, baseDelay: 100 });

    // Full jitter: Math.random() * baseDelay * 2^attempt
    // attempt 0: 0.5 * 100 * 1 = 50
    // attempt 1: 0.5 * 100 * 2 = 100
    expect(sleepCalls[0]).toBe(50);
    expect(sleepCalls[1]).toBe(100);

    randomSpy.mockRestore();
    vi.restoreAllMocks();
  });
});
```

Add a new constant after `SAFETY_MARGIN_SECONDS` (line 7):
```typescript
const MINIMUM_TTL_SECONDS = 10;
```

Replace line 122:
```typescript
const ttlSeconds = data.expires_in - SAFETY_MARGIN_SECONDS;
```
with:
```typescript
const ttlSeconds = Math.max(
  data.expires_in - SAFETY_MARGIN_SECONDS,
  MINIMUM_TTL_SECONDS,
```

This ensures that even when `expires_in` is 30 (30 - 60 = -30), the TTL floors at 10 seconds, preventing negative/zero expiresAt and the infinite auth refresh loop.
